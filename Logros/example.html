function query($db, $sql, $params, $return_status)
{
    $query = $db->prepare($sql);
    for ($i = 0; $i < sizeOf($params); $i++)
        $query->bindParam($i + 1, $params[$i]);
    $query->execute();
    if ($return_status) return $query
}
